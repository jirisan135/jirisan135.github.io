public class ExamConsole {

	}
	
	public void input() {
		Scanner scan = new Scanner(System.in);
		
		System.out.println("┌──────────────────────┐");
		System.out.println("│                       성적 입력	                         │");
		System.out.println("└──────────────────────┘");
		
		int kor;
		
		do {
			System.out.print("kor1 : ");
			kor = scan.nextInt();			
			if(kor < 0 || kor > 100) {
				System.out.println("kor1는 0~100 의 값을 입력해주세요.");
				kor = scan.nextInt();
			}
			
		} while (kor < 0 || kor > 100); 
		
		exam.setKor(kor);
		
		int eng;
		
		do {
			System.out.print("eng : ");
			eng = scan.nextInt();			
			if(eng < 0 || eng > 100) {
				System.out.println("eng는 0~100 의 값을 입력해주세요.");
				eng = scan.nextInt();
			}
			
		} while (eng < 0 || eng > 100); 
		
		exam.setEng(eng);
		
		int math;
		
		do {
			System.out.print("math : ");
			math = scan.nextInt();			
			if(math < 0 || math > 100) {
				System.out.println("math는 0~100 의 값을 입력해주세요.");
				math = scan.nextInt();
			}
			
		} while (math < 0 || math > 100); 
		
		exam.setMath(math);
		}
